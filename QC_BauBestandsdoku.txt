Private Sub TabOrtsbezug_DbrsASFOrtStrasse_OnDialogClosed(ByVal OKButton As Boolean)

	If Not OKButton Then Exit Sub
	If TabOrtsbezug_DbrsASFOrtStrasse.DataCount <> 1 Then Exit Sub

'*	Business Logik
'	Anschlussstellen einschr‰nken auf Straﬂenzug

	End With
	With TabOrtsbezug_DbrsASFOrtAStBis

		.RetrieveData
	End With
'	km-von und km-bis laden

	TabOrtsbezug_ASFDokStrKmBis.Data = TabOrtsbezug_DbrsASFOrtStrasse.Recordset.Fields("kmbis")

	TabOrtsbezug_DbrsASFOrtAStBis.ResetContent

		.KeyField = "ID"
		.RetrieveData
	End With

Private Sub TabOrtsbezug_DbrsASFDokObjekt_OnDialogClosed(ByVal OKButton As Boolean)

	If OKButton And TabOrtsbezug_DbrsASFDokObjekt.DataCount = 1 Then
		With TabOrtsbezug_DbrsASFDokObjekt.Recordset
			If Not .EOF Then
				'ObjektName
				TabOrtsbezug_ASFDokObjekt.Data = .Fields("Name")
				'Straﬂenzug
				If Not IsNull(.Fields("Straﬂenzug")) Then
					ASF_db_initDbrsWithRecordsetDefinition(TabOrtsbezug_DbrsASFOrtStrasse, "OrtsbezugStraﬂenz¸ge")
					TabOrtsbezug_DbrsASFOrtStrasse.ResetContent
					TabOrtsbezug_DbrsASFOrtStrasse.AddData .Fields("Straﬂenzug")
				End If
				'km von, ASt von
				If Not IsNull(.Fields("km von")) Then
					TabOrtsbezug_ASFDokStrKmVon.Data = .Fields("km von")
					TabOrtsbezug_DbrsASFOrtAStVon.ResetContent
				End If
				'km bis, ASt bis
				If Not IsNull(.Fields("km bis")) Then
					TabOrtsbezug_ASFDokStrKmBis.Data = .Fields("km bis")
					TabOrtsbezug_DbrsASFOrtAStBis.ResetContent
				End If
			End If
		End With
	End If

End Sub

Dim DOXiSCad As Object
Dim sAppTitle As String
Dim Rsp As Long
Dim buf As String

Private Function Initialize(lMode As Long) As Long
	sAppTitle = "DOXiS CAD Manager"
	On Error Resume Next
	Initialize = 0
	buf = DOXiSCad.Version
	If Err.Number  = 0 Then
		Exit Function
	End If
	Err.Clear
	Set DOXiSCad = CreateObject("DOXiSCad.Support")
	If Err Then
		MsgBox "Can't initialize DOXiSCad.Support!",vbExclamation , sAppTitle
		Initialize = -1
	End If
End Function

Private Sub Application_OnClose()
	If Initialize(0) <> 0 Then Exit Sub
	If Not DOXiSCad Is Nothing Then
		DOXiSCad.Quit
		Set DOXiSCad = Nothing
	End If
End Sub
